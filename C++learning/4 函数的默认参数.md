* # 默认形参
* 函数的形参可带有默认值。必须一律在最右边。

\#include <iostream>

using namespace std;



double test(double x, double y = 7)

{

 	return x - y;

}

 

int main()

{

 	cout << test(1, 3) << endl;

 	cout << test(15) << endl;



 	return 0;

}



运行结果：

-2

8
